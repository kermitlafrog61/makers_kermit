ctrl + alt + t - открыть терминал

ls - просмотр содержимого текущей директории
ls - путь_до_директории - просмотр содержимого указанной директории
ls -a - просмотр скрытых файлов директории

mkdir название_директории - создание директории
cd путь_до_директории - смена директории

pwd - вывод абсолютного пути в консоли
"." - ссылка на текущую директорию
".." - ссылка на директорию выше(родительскую)
"~" - ссылка на домашнюю директорию

абсолютный путь - путь, который начинается с корневой директории
(/home/username/...)
относительный путь - путь, отsносительно других директорий (.../some_dir)

nano путь_до_файла - открытие файла с помощью тектстового редактора,
если файла не существует создаёт его

mv путь_до_файла путь_до_директории - перенос директории/файла
в указанную директорию
mv путь_до_файла новое_название - переименование файла/директории

echo какой-то_текст - вывод указанного текста в консоль
echo some_text > new_file - запись/перезапись текста в файл
echo appended_text >> same_file - дозапись текста в файл

rm -  удаление файлов
rmdir - удаление директорий
rm -r - рекурсивно (удалить содержимое директории и ее саму)
rm -f - принудительное удаление
rm -rf - принудтельно + рекурсивно
